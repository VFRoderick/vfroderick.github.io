architecture-beta
    group azure(cloud)[Azure Cloud - ArcGIS Enterprise]

    group web(internet)[Web Tier] in azure
    group app(server)[Application Tier] in azure
    group data(database)[Data Tier] in azure

    service users(internet)[Users] in azure
    service lb(server)[Load Balancer] in web
    service wa(internet)[Web Adaptor] in web
    
    service portal(server)[Portal] in app
    service hosting(server)[Hosting Server x2] in app
    service mapping(server)[Mapping Server x4] in app
    service gp(server)[GP Server] in app
    service geoevent(server)[GeoEvent Server] in app
    
    service sql(database)[SQL Server] in data
    service datastore(disk)[ArcGIS Data Store] in data

    users:R --> L:lb
    lb:R --> L:wa
    wa:B --> T:portal
    portal:L --> R:hosting
    hosting:B --> T:mapping
    mapping:R --> L:gp
    gp:B --> T:geoevent
    portal:B --> T:sql
    sql:R --> L:datastore
    hosting:B --> T:datastore
